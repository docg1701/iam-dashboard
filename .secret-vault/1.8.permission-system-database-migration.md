## Problem Statement

6. **Migration Validation:** Comprehensive validation ensures migration completed successfully
7. **Performance Optimization:** New database indexes ensure permission checking remains fast
8. **Backup and Recovery:** Complete backup procedures before migration with recovery testing

### Data Migration Logic

**Default Permission Assignment:**
```python
async def migrate_existing_users():
    """Migrate existing users to new permission system."""
    
    # Get system user for created_by field
    system_user = await get_system_user()
    
    # Get all existing users
    users = await db.execute(select(User))
    
    for user in users.scalars():
        if user.role == "sysadmin":
            # Sysadmin gets full access to all agents
            permissions = {
                "client_management": {"create": True, "read": True, "update": True, "delete": True},
                "pdf_processing": {"create": True, "read": True, "update": True, "delete": True},
                "reports_analysis": {"create": True, "read": True, "update": True, "delete": True},
                "audio_recording": {"create": True, "read": True, "update": True, "delete": True}
            }
            
        elif user.role == "admin":
            # Admin gets access to client management and reports
            permissions = {
                "client_management": {"create": True, "read": True, "update": True, "delete": True},
                "reports_analysis": {"create": True, "read": True, "update": True, "delete": False}
            }
            
        else:  # user role
            # Regular users get read-only access to client management
            permissions = {
                "client_management": {"create": False, "read": True, "update": False, "delete": False}
            }
        
        # Create permission records for each agent
        for agent_name, agent_permissions in permissions.items():
            permission_record = UserAgentPermission(
                user_id=user.user_id,
                agent_name=agent_name,
                permissions=agent_permissions,
                created_by_user_id=system_user.user_id
            )
            db.add(permission_record)
    
    await db.commit()
    
    # Validate migration
    await validate_permission_migration()

async def validate_permission_migration():
    """Validate that all users have appropriate permissions after migration."""
    
    # Check that all users have permission records
    users_without_permissions = await db.execute(
        select(User.user_id)
        .outerjoin(UserAgentPermission)
        .where(UserAgentPermission.user_id.is_(None))
    )
    
    if users_without_permissions.scalars().first():
        raise MigrationError("Some users are missing permission records")
    
    # Check permission integrity
    invalid_permissions = await db.execute(
        select(UserAgentPermission)
        .where(UserAgentPermission.permissions == {})
    )
    
    if invalid_permissions.scalars().first():
        raise MigrationError("Found permission records with empty permissions")
    
    print("✅ Permission migration validation completed successfully")
```

        sa.Column('user_id', postgresql.UUID(as_uuid=True), nullable=False),
        sa.Column('agent_name', sa.VARCHAR(length=50), nullable=False),
        sa.Column('permissions', postgresql.JSONB(astext_type=sa.Text()), nullable=False, default=sa.text("'{}'::jsonb")),
        sa.Column('created_by_user_id', postgresql.UUID(as_uuid=True), nullable=False),
        sa.Column('created_at', sa.TIMESTAMP(timezone=True), nullable=False, default=sa.text('NOW()')),

        sa.PrimaryKeyConstraint('permission_id'),

        sa.ForeignKeyConstraint(['created_by_user_id'], ['users.user_id']),
        sa.CheckConstraint(
            "agent_name IN ('client_management', 'pdf_processing', 'reports_analysis', 'audio_recording')",
            name='chk_agent_name_valid'
        ),
        sa.CheckConstraint(
            "permissions ? 'create' OR permissions ? 'read' OR permissions ? 'update' OR permissions ? 'delete'",
            name='chk_permissions_structure'
        ),
        sa.UniqueConstraint('user_id', 'agent_name', name='uk_user_agent_permissions_user_agent')
    )
    
    # Create indexes
    op.create_index('idx_user_agent_permissions_user_id', 'user_agent_permissions', ['user_id'])
    op.create_index('idx_user_agent_permissions_agent_name', 'user_agent_permissions', ['agent_name'])
    op.create_index('idx_user_agent_permissions_created_by', 'user_agent_permissions', ['created_by_user_id'])
    op.create_index('idx_user_agent_permissions_permissions_gin', 'user_agent_permissions', ['permissions'], postgresql_using='gin')
    op.create_index('idx_user_agent_permissions_user_agent', 'user_agent_permissions', ['user_id', 'agent_name'])
    
    # Create system user for migration operations if it doesn't exist
    op.execute(text("""
        INSERT INTO users (user_id, email, password_hash, full_name, role, is_active)
        SELECT 
            uuid_generate_v4(),
            'system@migration.local',
            '$2b$12$dummy.hash.for.system.user.migration',
            'System Migration User',
            'sysadmin',
            false
        WHERE NOT EXISTS (
            SELECT 1 FROM users WHERE email = 'system@migration.local'
        );
    """))
    
    # Migrate existing users to new permission system
    op.execute(text("""
        WITH system_user AS (
            SELECT user_id FROM users WHERE email = 'system@migration.local'
        ),
        user_permissions AS (
            SELECT 
                u.user_id,
                u.role,
                s.user_id as system_user_id,
                CASE 
                    WHEN u.role = 'sysadmin' THEN 
                        jsonb_build_object('create', true, 'read', true, 'update', true, 'delete', true)
                    WHEN u.role = 'admin' THEN 
                        jsonb_build_object('create', true, 'read', true, 'update', true, 'delete', true)
                    ELSE 
                        jsonb_build_object('create', false, 'read', true, 'update', false, 'delete', false)
                END as permissions
            FROM users u, system_user s
            WHERE u.email != 'system@migration.local'
        )
        INSERT INTO user_agent_permissions (user_id, agent_name, permissions, created_by_user_id)
        SELECT 
            up.user_id,
            agent_name,
            CASE 
                WHEN up.role = 'sysadmin' THEN up.permissions
                WHEN up.role = 'admin' AND agent_name IN ('client_management', 'reports_analysis') THEN up.permissions
                WHEN up.role = 'user' AND agent_name = 'client_management' THEN up.permissions
                ELSE jsonb_build_object('create', false, 'read', false, 'update', false, 'delete', false)
            END as permissions,
            up.system_user_id
        FROM user_permissions up
        CROSS JOIN (
            VALUES 
                ('client_management'),
                ('pdf_processing'),
                ('reports_analysis'),
                ('audio_recording')
        ) AS agents(agent_name);
    """))

async def validate_pre_migration():
    """Validate database state before migration."""
    
    # Check for required extensions
    result = await db.execute(text("SELECT 1 FROM pg_extension WHERE extname = 'uuid-ossp'"))
    if not result.scalar():
        raise MigrationError("uuid-ossp extension is required")
    
    # Check users table exists and has expected columns
    result = await db.execute(text("""
        SELECT column_name 
        FROM information_schema.columns 
        WHERE table_name = 'users' 
        AND column_name IN ('user_id', 'role', 'email')
    """))
    
    if len(result.fetchall()) != 3:
        raise MigrationError("Users table missing required columns")
    
    # Check for existing user data
    user_count = await db.execute(text("SELECT COUNT(*) FROM users"))
    if user_count.scalar() == 0:
        raise MigrationError("No users found - cannot proceed with migration")
    
    print("✅ Pre-migration validation completed successfully")

async def validate_post_migration():
    """Validate database state after migration."""
    
    # Check table was created with correct structure
    result = await db.execute(text("""
        SELECT COUNT(*) 
        FROM information_schema.tables 
        WHERE table_name = 'user_agent_permissions'
    """))
    
    if result.scalar() != 1:
        raise MigrationError("user_agent_permissions table was not created")
    
    # Check all users have permission records
    result = await db.execute(text("""
        SELECT u.user_id 
        FROM users u 
        LEFT JOIN user_agent_permissions uap ON u.user_id = uap.user_id 
        WHERE uap.user_id IS NULL 
        AND u.email != 'system@migration.local'
    """))
    
    if result.fetchall():
        raise MigrationError("Some users are missing permission records")
    
    # Check permission integrity
    result = await db.execute(text("""
        SELECT COUNT(*) 
        FROM user_agent_permissions 
        WHERE permissions = '{}'::jsonb
    """))
    
    if result.scalar() > 0:
        raise MigrationError("Found invalid empty permission records")
    
    print("✅ Post-migration validation completed successfully")

```python
class MigrationMonitor:
    def __init__(self, db: AsyncSession, redis: Redis):
        self.db = db
        self.redis = redis
        self.migration_id = "user_agent_permissions_2025080401"
    
    async def start_migration(self):
        """Start migration monitoring."""
        await self.redis.hset(
            f"migration:{self.migration_id}",
            mapping={
                "status": "in_progress",
                "started_at": datetime.utcnow().isoformat(),
                "progress": "0"
            }
        )
    
    async def update_progress(self, step: str, progress: int):
        """Update migration progress."""
        await self.redis.hset(
            f"migration:{self.migration_id}",
            mapping={
                "current_step": step,
                "progress": str(progress),
                "updated_at": datetime.utcnow().isoformat()
            }
        )
    
    async def complete_migration(self, success: bool):
        """Mark migration as completed."""
        status = "completed" if success else "failed"
        await self.redis.hset(
            f"migration:{self.migration_id}",
            mapping={
                "status": status,
                "completed_at": datetime.utcnow().isoformat(),
                "progress": "100" if success else "error"
            }
        )

async def execute_monitored_migration():
    """Execute migration with monitoring."""
    monitor = MigrationMonitor(db, redis)
    
    try:
        await monitor.start_migration()
        
        await monitor.update_progress("pre_validation", 10)
        await validate_pre_migration()
        
        await monitor.update_progress("schema_creation", 30)
        # Execute Alembic migration
        
        await monitor.update_progress("data_migration", 60)
        await migrate_existing_users()
        
        await monitor.update_progress("post_validation", 90)
        await validate_post_migration()
        
        await monitor.complete_migration(True)
        
    except Exception as e:
        await monitor.complete_migration(False)
        raise MigrationError(f"Migration failed: {str(e)}")
```
