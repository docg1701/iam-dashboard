## Story

## Problem Statement
The transition from a simple 3-role system to a complex agent-based permission system introduces significant complexity that must be thoroughly tested. We need comprehensive test coverage, performance validation, security testing, and user acceptance testing to ensure the new system works reliably in production.

## Acceptance Criteria

1. **Permission Logic Testing:** Comprehensive unit and integration tests covering all permission scenarios and edge cases
2. **API Security Testing:** Validation that permission middleware correctly blocks unauthorized access to all endpoints
3. **Frontend Permission Testing:** UI components properly show/hide based on user permissions with no security bypasses
4. **Performance Testing:** Permission checking performs within acceptable limits under realistic load conditions
5. **Migration Testing:** Database migration works correctly with various data scenarios and can be safely rolled back
6. **End-to-End User Workflows:** Complete user journey testing for different permission configurations
7. **Security Penetration Testing:** Attempt to bypass permission controls and validate system security boundaries
8. **User Acceptance Testing:** Real users validate that the new permission system meets their workflow needs

- **Integration Tests (30%):** API and database interaction testing

```python
class TestPermissionService:
    
    @pytest.mark.asyncio
    async def test_has_agent_permission_sysadmin_always_allowed(
        self, permission_service, sysadmin_user
    ):
        """Test that sysadmin always has access to all operations."""
        # Test all agents and operations
        for agent in AGENT_TYPES:
            for operation in OPERATIONS:
                result = await permission_service.has_agent_permission(
                    sysadmin_user.user_id, agent, operation
                )
                assert result is True, f"Sysadmin should have {operation} access to {agent}"
    
    @pytest.mark.asyncio
    async def test_has_agent_permission_regular_user_specific_permissions(
        self, permission_service, regular_user, db
    ):
        """Test that regular users only have explicitly granted permissions."""
        # Grant specific permission
        await permission_service.assign_agent_permissions(
            user_id=regular_user.user_id,
            agent_name="client_management",
            permissions={"read": True, "create": False, "update": False, "delete": False},
            assigned_by=sysadmin_user.user_id
        )
        
        # Test granted permission
        result = await permission_service.has_agent_permission(
            regular_user.user_id, "client_management", "read"
        )
        assert result is True
        
        # Test denied permission
        result = await permission_service.has_agent_permission(
            regular_user.user_id, "client_management", "create"
        )
        assert result is False
        
        # Test no permission for other agents
        result = await permission_service.has_agent_permission(
            regular_user.user_id, "pdf_processing", "read"
        )
        assert result is False
    
    @pytest.mark.asyncio
    async def test_permission_caching_and_invalidation(
        self, permission_service, regular_user, redis_mock
    ):
        """Test that permission caching works correctly."""
        # First call should query database and cache result
        result1 = await permission_service.has_agent_permission(
            regular_user.user_id, "client_management", "read"
        )
        
        # Verify cache was populated
        cache_key = f"permissions:{regular_user.user_id}:client_management"
        cached_data = await redis_mock.get(cache_key)
        assert cached_data is not None
        
        # Update permissions
        await permission_service.assign_agent_permissions(
            user_id=regular_user.user_id,
            agent_name="client_management", 
            permissions={"read": True, "create": True},
            assigned_by=sysadmin_user.user_id
        )
        
        # Verify cache was invalidated
        cached_data = await redis_mock.get(cache_key)
        assert cached_data is None
        
        # New call should reflect updated permissions
        result2 = await permission_service.has_agent_permission(
            regular_user.user_id, "client_management", "create"
        )
        assert result2 is True
```

```typescript
      'create'
    );
  });
  
  it('should render fallback when user lacks required permission', () => {
    const mockPermissions = {
      hasAgentPermission: vi.fn().mockReturnValue(false)
    };
    
    vi.mocked(useUserPermissions).mockReturnValue(mockPermissions);
    
    render(
      <PermissionGuard 
        agent="client_management" 
        operation="create"
        fallback={<div>Access Denied</div>}
      >
        <button>Create Client</button>
      </PermissionGuard>
    );
    
    expect(screen.queryByRole('button')).not.toBeInTheDocument();
    expect(screen.getByText('Access Denied')).toBeInTheDocument();
  });
  
  it('should update visibility when permissions change', async () => {
    const mockPermissions = {
      hasAgentPermission: vi.fn().mockReturnValue(false)
    };
    
    const { rerender } = render(
      <PermissionGuard agent="client_management" operation="create">
        <button>Create Client</button>
      </PermissionGuard>
    );
    
    // Initially no permission
    expect(screen.queryByRole('button')).not.toBeInTheDocument();
    
    // Update permissions
    mockPermissions.hasAgentPermission.mockReturnValue(true);
    
    rerender(
      <PermissionGuard agent="client_management" operation="create">
        <button>Create Client</button>
      </PermissionGuard>
    );
    
    // Now should be visible
    expect(screen.getByRole('button', { name: /create client/i })).toBeInTheDocument();
```

```python
class TestPermissionPerformance:
    
    @pytest.mark.asyncio
    async def test_permission_checking_performance(
        self, permission_service, users_dataset_100
    ):
        """Test permission checking performance with 100 users."""
        start_time = time.perf_counter()
        
        # Check permissions for all users
        tasks = []
        for user in users_dataset_100:
            task = permission_service.has_agent_permission(
                user.user_id, "client_management", "read"
            )
            tasks.append(task)
        
        results = await asyncio.gather(*tasks)
        
        end_time = time.perf_counter()
        total_time = end_time - start_time
        
        # Should complete within 1 second for 100 users
        assert total_time < 1.0, f"Permission checking took {total_time:.2f}s for 100 users"
        
        # All results should be boolean
        assert all(isinstance(result, bool) for result in results)
    
    @pytest.mark.asyncio
    async def test_redis_caching_effectiveness(
        self, permission_service, regular_user, redis_client
    ):
        """Test that Redis caching improves performance."""
        
        # Clear cache
        await redis_client.flushdb()
        
        # First call (no cache) - measure time
        start_time = time.perf_counter()
        await permission_service.has_agent_permission(
            regular_user.user_id, "client_management", "read"
        )
        uncached_time = time.perf_counter() - start_time
        
        # Second call (cached) - measure time
        start_time = time.perf_counter()
        await permission_service.has_agent_permission(
            regular_user.user_id, "client_management", "read"
        )
        cached_time = time.perf_counter() - start_time
        
        # Cached call should be significantly faster
        assert cached_time < uncached_time * 0.5, "Caching should improve performance by 50%"
```

```python
class TestPermissionSecurity:
    
    @pytest.mark.asyncio
    async def test_cannot_bypass_permissions_with_direct_api_calls(
        self, test_client, regular_user_headers
    ):
        """Test that direct API calls cannot bypass permission checks."""
        
        # Try to create client without permission
        response = await test_client.post(
            "/api/v1/clients/",
            json={"full_name": "Bypass Test", "ssn": "999-99-9999", "birth_date": "1990-01-01"},
            headers=regular_user_headers
        )
        assert response.status_code == 403
        
        # Try to access admin endpoints
        response = await test_client.get(
            "/api/v1/admin/users",
            headers=regular_user_headers
        )
        assert response.status_code == 403
        
        # Try to modify other user's permissions
        response = await test_client.put(
            "/api/v1/users/some-other-user-id/permissions",
            json={"client_management": {"create": True}},
            headers=regular_user_headers
        )
        assert response.status_code == 403
    
    @pytest.mark.asyncio
    async def test_permission_changes_invalidate_sessions(
        self, test_client, regular_user, admin_user_headers
    ):
        """Test that permission changes affect user immediately."""
        
        # User initially has no permissions
        user_headers = await get_auth_headers(regular_user)
        
        response = await test_client.post(
            "/api/v1/clients/",
            json={"full_name": "Test", "ssn": "123-45-6789", "birth_date": "1990-01-01"},
            headers=user_headers
        )
        assert response.status_code == 403
        
        # Admin grants permission
        await test_client.put(
            f"/api/v1/users/{regular_user.user_id}/permissions",
            json={
                "client_management": {"create": True, "read": True, "update": False, "delete": False}
            },
            headers=admin_user_headers
        )
        
        # User should now have access (real-time update)
        response = await test_client.post(
            "/api/v1/clients/",
            json={"full_name": "Test", "ssn": "123-45-6789", "birth_date": "1990-01-01"},
            headers=user_headers
        )
        assert response.status_code == 201
```
