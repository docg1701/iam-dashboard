### Data Models
**Migration Target Schema** [Source: architecture/database-schema.md#database-migration-strategy]
```sql
-- Primary permission table
CREATE TABLE user_agent_permissions (
    permission_id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    user_id UUID NOT NULL REFERENCES users(user_id) ON DELETE CASCADE,
    agent_name VARCHAR(50) NOT NULL CHECK (agent_name IN (
        'client_management', 'pdf_processing', 'reports_analysis', 'audio_recording'
    )),
    permissions JSONB NOT NULL DEFAULT '{}',
    created_by_user_id UUID NOT NULL REFERENCES users(user_id),
    created_at TIMESTAMP WITH TIME ZONE NOT NULL DEFAULT NOW(),
    updated_at TIMESTAMP WITH TIME ZONE NOT NULL DEFAULT NOW(),
    
    UNIQUE(user_id, agent_name),
    CONSTRAINT permissions_jsonb_structure CHECK (
        jsonb_typeof(permissions) = 'object' AND
        permissions ? 'create' AND jsonb_typeof(permissions->'create') = 'boolean' AND
        permissions ? 'read' AND jsonb_typeof(permissions->'read') = 'boolean' AND
        permissions ? 'update' AND jsonb_typeof(permissions->'update') = 'boolean' AND
        permissions ? 'delete' AND jsonb_typeof(permissions->'delete') = 'boolean'
    )
);

-- Permission templates for role-based assignments
CREATE TABLE permission_templates (
    template_id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    template_name VARCHAR(100) NOT NULL UNIQUE,
    description TEXT,
    permissions JSONB NOT NULL,
    is_system_template BOOLEAN NOT NULL DEFAULT false,
    created_by_user_id UUID NOT NULL REFERENCES users(user_id),
    created_at TIMESTAMP WITH TIME ZONE NOT NULL DEFAULT NOW(),
    updated_at TIMESTAMP WITH TIME ZONE NOT NULL DEFAULT NOW()
);

-- Permission change audit log
CREATE TABLE permission_audit_log (
    audit_id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    user_id UUID NOT NULL REFERENCES users(user_id),
    agent_name VARCHAR(50) NOT NULL,
    action VARCHAR(20) NOT NULL CHECK (action IN ('GRANT', 'REVOKE', 'UPDATE', 'BULK_GRANT', 'BULK_REVOKE', 'TEMPLATE_APPLIED')),
    old_permissions JSONB,
    new_permissions JSONB,
    changed_by_user_id UUID NOT NULL REFERENCES users(user_id),
    change_reason TEXT,
    ip_address INET,
    user_agent TEXT,
    timestamp TIMESTAMP WITH TIME ZONE NOT NULL DEFAULT NOW()
);
```

def upgrade():
    # Phase 1: Create tables with constraints
    op.create_table('user_agent_permissions', ...)
    op.create_table('permission_templates', ...)
    op.create_table('permission_audit_log', ...)
    
    # Phase 2: Create performance indexes concurrently
    op.create_index_concurrently('idx_user_agent_permissions_composite', ...)
    op.create_index_concurrently('idx_user_agent_permissions_jsonb', ...)
    
    # Phase 3: Create database functions
    op.execute('CREATE OR REPLACE FUNCTION check_user_agent_permission(...)')
    op.execute('CREATE OR REPLACE FUNCTION log_permission_change(...)')
    
    # Phase 4: Migrate existing user permissions
    op.execute("""
        WITH system_user AS (INSERT INTO users (...) RETURNING user_id),
        user_permissions AS (SELECT user_id, role, ... FROM users)
        INSERT INTO user_agent_permissions (user_id, agent_name, permissions, created_by_user_id)
        SELECT ... FROM user_permissions CROSS JOIN agents
    """)
    
    # Phase 5: Create system permission templates
    op.execute("INSERT INTO permission_templates (...) VALUES (...)")

def downgrade():
    # Safe rollback procedure
    op.create_table('user_agent_permissions_backup', ...)
    op.execute('INSERT INTO user_agent_permissions_backup SELECT * FROM user_agent_permissions')
    op.drop_table('permission_audit_log')
    op.drop_table('user_agent_permissions')

### Technical Constraints
**Migration Performance Requirements** [Source: architecture/database-schema.md#migration-performance-monitoring]
- **Migration Execution Time**: Complete migration for 100 users in <5 minutes
- **Permission Query Performance**: Sub-200ms average, <50ms for cached queries
- **Database Locking**: Minimize table locking during migration, use concurrent index creation
- **Memory Usage**: Migration process should not exceed 512MB memory consumption
- **Disk Space**: Migration requires <100MB additional disk space for indexes and audit logs

**Migration Safety Requirements** [Source: architecture/database-schema.md#rollback-procedures]
- **Zero Data Loss**: All existing user data must be preserved during migration
- **Backward Compatibility**: Existing API endpoints continue to work during migration
- **Rollback Capability**: Complete rollback possible within 5 minutes
- **Atomic Operations**: Migration uses database transactions for consistency
- **Progress Monitoring**: Real-time migration progress tracking and health checks
