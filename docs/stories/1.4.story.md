```python
@app.websocket("/ws/admin/agent-executions")
async def agent_execution_websocket(websocket: WebSocket):
    await websocket.accept()
    while True:
        execution_statuses = await get_all_execution_statuses()
        await websocket.send_json(execution_statuses)
        await asyncio.sleep(1)

# UI Integration with WebSocket updates
ui.timer(1.0, lambda: update_execution_status(), once=False)

async def update_execution_status():
    """Update agent execution status via WebSocket connection"""
    try:
        websocket = await connect_websocket("/ws/admin/agent-executions")
        execution_data = await websocket.receive_json()
        # Update UI components with real-time data
        update_execution_cards(execution_data)
    except WebSocketException:
        # Fallback to polling if WebSocket fails
        execution_data = await fetch_execution_status_http()
        update_execution_cards(execution_data)
```
