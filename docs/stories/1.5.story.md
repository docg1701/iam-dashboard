async def execute_agent(agent_id: str, execution_request: AgentExecutionRequest, current_user: dict = Depends(AuthManager.get_current_user)):
    try:
        agent = await agent_manager.get_agent(agent_id)
        if not agent:

        raise HTTPException(status_code=500, detail=str(e))

### Component Specifications
**Document Processing Service Layer** [Source: architecture/component-architecture-and-integration-patterns.md#4.2]:
```python
class DocumentProcessingService:
    def __init__(self, agent_manager: AgentManager, document_repository: DocumentRepository):
        self.agent_manager = agent_manager
        self.document_repository = document_repository
        self.execution_tracker = ExecutionTracker()
    
    async def process_document(self, file_data: bytes, filename: str, client_id: int, user_id: int) -> ProcessingResult:
        execution = await self.execution_tracker.start_execution(
            agent_id="pdf_processor", user_id=user_id, client_id=client_id,
            input_data={"filename": filename, "file_size": len(file_data)}
        )
        
        try:
            agent = await self.agent_manager.get_agent("pdf_processor")
            if not agent:
                agent = await self.agent_manager.create_agent("pdf_processor", "default_pdf_processor")
            
            processing_result = await agent.process({
                "file_data": file_data, "filename": filename, "client_id": client_id
            })
            
            document = await self.document_repository.create({
                "filename": filename,  # NOT original_filename
                "client_id": client_id,
                "agent_execution_id": execution.execution_id,
                "file_path": processing_result["file_path"],
                "file_size": len(file_data),
                "mime_type": processing_result["mime_type"],
                "status": DocumentStatus.PROCESSED,  # NOT processing_status
                "document_type": processing_result.get("document_type", "simple"),
                "content_hash": processing_result["content_hash"]
                # NO embedding_vector field - embeddings go in DocumentChunk table
            })
            
            # Store embeddings in DocumentChunk table (768 dimensions)
            if processing_result.get("chunks"):
                for chunk in processing_result["chunks"]:
                    await self.chunk_repository.create({
                        "document_id": document.id,
                        "node_id": chunk["node_id"],
                        "embedding": chunk["embedding"],  # 768 dims from Gemini
                        "text": chunk["text"],
                        "chunk_metadata": chunk["metadata"]
                    })
            
            await self.execution_tracker.complete_execution(
                execution.execution_id, success=True,
                output_data={"document_id": document.document_id}
            )
            
            return ProcessingResult(success=True, document_id=document.document_id,
                                  extracted_text=processing_result["extracted_text"])
            
        except Exception as e:
            await self.execution_tracker.complete_execution(
                execution.execution_id, success=False, error_message=str(e)
            )
            return ProcessingResult(success=False, error_message=str(e))
```

**NiceGUI Document Upload Component** [Source: architecture/user-interface-integration.md#6]:
```python
class DocumentUploadComponent:
    def __init__(self):
        self.document_service = DocumentProcessingService()
        self.processing_status = {}
    
    def create(self) -> None:
        with ui.card().classes('w-full p-6'):
            ui.label('Processamento de Documentos').classes('text-xl font-semibold mb-4')
            
            # Drag and drop upload area
            upload = ui.upload(
                on_upload=self._handle_pdf_upload,
                multiple=False,
                auto_upload=True
            ).classes('w-full').props('accept=".pdf" drag-over-text="Solte o PDF aqui" label="Clique ou arraste um PDF"')
            
            # Progress tracking area
            self.progress_container = ui.column().classes('w-full mt-4')
    
    async def _handle_pdf_upload(self, upload_event):
        try:
            current_user = AuthManager.get_current_user()
            client_id = 1  # Get from context
            
            # Start processing with progress tracking
            document_id = str(uuid4())
            self.processing_status[document_id] = {"status": "uploading", "progress": 0}
            
            # Create progress display
            with self.progress_container:
                progress_card = ui.card().classes('w-full p-4')
                with progress_card:
                    ui.label(f'Processando: {upload_event.name}').classes('font-medium')
                    progress_bar = ui.linear_progress(value=0).classes('w-full mt-2')
                    status_label = ui.label('Iniciando processamento...').classes('text-sm text-gray-600 mt-2')
            
            # Start WebSocket for progress updates
            self._start_progress_websocket(document_id, progress_bar, status_label)
            
            # Process document
            result = await self.document_service.process_document(
                file_data=upload_event.content.read(),
                filename=upload_event.name,
                client_id=client_id,
                user_id=current_user["id"]
            )
            
            if result.success:
                # Show results
                await self._display_processing_results(result, upload_event.name)
                ui.notify("Documento processado com sucesso!", type="positive")
            else:
                ui.notify(f"Erro no processamento: {result.error_message}", type="negative")
                
        except Exception as e:
            ui.notify(f"Erro: {str(e)}", type="negative")
    
    async def _display_processing_results(self, result: ProcessingResult, filename: str):
        with self.progress_container:
            results_card = ui.card().classes('w-full p-4 mt-4')
            with results_card:
                ui.label(f'Resultados: {filename}').classes('text-lg font-semibold mb-3')
                
                # Extracted text preview
                ui.label('Texto Extraído:').classes('font-medium')
                ui.textarea(value=result.extracted_text[:500] + "..." if len(result.extracted_text) > 500 else result.extracted_text).classes('w-full').props('readonly')
                
                # Document insights
                if result.insights:
                    ui.label('Insights Gerados pela IA:').classes('font-medium mt-4')
                    for insight in result.insights:
                        ui.markdown(f"• **{insight['category']}**: {insight['content']}")
```
